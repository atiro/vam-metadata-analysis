#    print(dept_col_df)
    if(len(dept_col_df) < 5):
        continue
    
    chart = alt.Chart(dept_col_df).mark_bar().encode(
      x=alt.X('index:O', axis=alt.Axis(title='Cardinality')),
      y=alt.Y(column, type='quantitative',  axis=alt.Axis(title='Objects'))
    ).properties(title=column.split(":")[1])
    charts.append(chart)
    if count > 1:
      hcharts.append(alt.hconcat(*charts))
      charts = []
      count = 0
    else:
      count += 1
