# Now go through each with lowercase, and check if it also exists in original list (upper case first letter for the comparision) 

# Index - Singular Name, Count, Plural name, count
duplicates = pd.DataFrame()
total_lower = 0
total_noupper = 0

data = []
data_noupper = []
for index, value in lower_types.items():
    upper = index.capitalize()
    if upper in obj_types:
        count = obj_types.get(upper)
        total_lower += value
        data.append([upper, count, index, value])
    else:
        # Even stranger - no uppercase version exists - probably a typo
        total_noupper += value
        data_noupper.append([index, value])

duplicates = pd.DataFrame(data, columns = ["Uppercase Name", "Uppercase Objects", "lowercase Name", "lowercase Count"])
     
noupper = pd.DataFrame(data_noupper, columns = ["lowercase Name", "lowercase Count"])
