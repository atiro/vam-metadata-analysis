# Now go through each, remove the 's', and check if it also exists in original list (upper case first letter for the comparision) 

# Index - Singular Name, Count, Plural name, count
duplicates = pd.DataFrame()
total_plural = 0

data = []
for index, value in plural_types.items():
    singular = index[:-1].capitalize()
    if singular in obj_types:
        count = obj_types.get(singular)
        data.append([singular, count, index, value])
        total_plural += value

duplicates = pd.DataFrame(data, columns = ["Singular Name", "Singular Objects", "Plural Name", "Plural Count"])

print(f"Total Objects {total_objects} of which {total_plural} ({(total_plural/total_objects)*100:.4f}%) are possible plural duplicates")
