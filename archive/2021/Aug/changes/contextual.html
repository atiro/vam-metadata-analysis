import altair as alt

vcharts = []
hcharts = []
charts = []
dept = None
indexn = None
count = 0
for index, row in percentage_df.iterrows():
    dept = row.to_frame(name='concepts')
    charts.append(
        alt.Chart(dept.reset_index()).mark_bar().encode(
        x=alt.X('index:N', axis=alt.Axis(title='Cataloguing Area')),
        y=alt.Y('concepts',  axis=alt.Axis(format='%', title='Completeness'))
       ).properties(title=index))
    if count > 3:
      hcharts.append(alt.hconcat(*charts))
      charts = []
      count = 0
    else:
      count += 1

alt.vconcat(*hcharts)
